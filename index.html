-- ============================================================ -- AI 활용동호회 : PC 통신 — 가입 신청 테이블 -- Supabase 대시보드 > SQL Editor 에 붙여넣고 실행하세요. -- ============================================================ create table if not exists club_signups ( id uuid primary key default gen_random_uuid(), created_at timestamptz default now(), handle text not null, -- 대화명(닉네임) field text, -- 하는 일 / 분야 tools text[], -- 써본 AI 도구 (복수) level text, -- 활용 수준 pain_point text, -- AI로 풀고 싶은 고민 interests text[], -- 배우고 싶은 주제 (복수) meeting_style text, -- 선호 모임 방식 message text -- 운영진에게 한마디 ); -- RLS 활성화 후, 익명 사용자의 "삽입(가입신청)"만 허용 alter table club_signups enable row level security; create policy "anyone can sign up" on club_signups for insert to anon with check (true); -- (선택) 가입 현황을 한눈에 보는 뷰 create or replace view club_signups_summary as select created_at::date as day, count(*) as signups from club_signups group by 1 order by 1 desc;
